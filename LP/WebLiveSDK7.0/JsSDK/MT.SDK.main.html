Example
var TF = new MT.SDK.main(access_token, {
    config: {
       techOrder: 'FLV' || 'HLS' || 'FLASH', // 流媒体格式
       controls: true // 播放器的控件（全屏等）
    },
    video: {
      autoplay: true, // 自动播放
      controls: false, // 媒体原生控件
      preload: 'auto', // 预加载
      poster: false, // 海报
     }
  }, function (initData) {
     todo(...)
  });
